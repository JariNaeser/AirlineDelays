df["CANCELLED"] = pd.to_numeric(df["CANCELLED"], errors="coerce")
cancelledFlightsPerAirline = df[df["CANCELLED"] == 1]["FULL_AIRLINE_NAME"].value_counts().reset_index()
totalFlightsPerAirline = df[df["CANCELLED"] == 0]["FULL_AIRLINE_NAME"].value_counts().reset_index()

df_merged = pd.merge(totalFlightsPerAirline, cancelledFlightsPerAirline, on='FULL_AIRLINE_NAME')
df_merged.columns = ["FULL_AIRLINE_NAME", "TOTAL_FLIGHTS", "TOTAL_CANCELLATIONS"]



df_merged["CANC_INTERVAL"] = (df_merged["TOTAL_CANCELLATIONS"] / df_merged["TOTAL_FLIGHTS"]) * 100

# Divide total flights per cancelled ones for each airline
fig = px.histogram(df_merged, y="FULL_AIRLINE_NAME", x="CANC_INTERVAL")
fig.update_layout(xaxis_title="Cancellation probability [%]", yaxis_title="Airline name", showlegend=False)
fig.update_yaxes(categoryorder="total descending")
fig.show(renderer="png")

dfFiltered = df[(df['DEPARTURE_DELAY'] <= 800) & (df['ARRIVAL_DELAY'] <= 800)]

fig_scatter = px.scatter(dfFiltered, x="DEPARTURE_DELAY", y="ARRIVAL_DELAY",
                         width=800, height=600, opacity=0.05)
fig_scatter.update_layout(xaxis_title="Departure delay [minutes]", yaxis_title="Arrival delay [minutes]")
fig_scatter.update_yaxes(range=[-100, 800])
fig_scatter.update_xaxes(range=[-100, 800])
fig_scatter.update_traces(marker=dict(size=3))


interval_box = 30
df['DEPARTURE_DELAY_BIN_BOX'] = pd.cut(df['DEPARTURE_DELAY'], bins=range(-100, 800 + interval_box, interval_box), right=False)
df = df.dropna(subset=['DEPARTURE_DELAY_BIN_BOX'])
df['DEPARTURE_DELAY_BIN_BOX'] = df['DEPARTURE_DELAY_BIN_BOX'].apply(lambda x: x.mid)

fig_box = px.box(df, x='DEPARTURE_DELAY_BIN_BOX', y='ARRIVAL_DELAY', orientation='v')

fig_box.update_layout(xaxis_title="Departure delay [minutes]", yaxis_title="Arrival delay [minutes]")
fig_box.update_yaxes(range=[-100, 800])
fig_box.update_xaxes(range=[-100, 800])
fig_box.update_yaxes(showticklabels=False)


fig = make_subplots(rows=1, cols=2,shared_yaxes=False, shared_xaxes=False, x_title='Departure delay [minutes]',
                    y_title='Arrival delay [minutes]')

df['DATE'] = pd.to_datetime(df[['YEAR', 'MONTH', 'DAY']])

df['DAY_NAME'] = df['DATE'].dt.day_name()
df['WEEK_NUMBER'] = df['DATE'].dt.isocalendar().week
flights_per_weekday_week = df.groupby(['DAY_NAME', 'WEEK_NUMBER']).size().reset_index(name='FLIGHT_COUNT')

weekdays_order = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"]
flights_per_weekday_week['DAY_NAME'] = pd.Categorical(flights_per_weekday_week['DAY_NAME'], categories=weekdays_order, ordered=True)

heatmap_data = flights_per_weekday_week.pivot(index='DAY_NAME', columns='WEEK_NUMBER', values='FLIGHT_COUNT')

fig = px.imshow(heatmap_data,
                labels=dict(x="Week Number", y="Day of Week", color="Number of Flights"),
                x=heatmap_data.columns,
                y=heatmap_data.index,
                color_continuous_scale="Viridis",  
                title="Number of Flights per Weekday and Week Number",
                )

fig.show(renderer="png")

departures_per_airport = df["ORIGIN_AIRPORT"].value_counts().reset_index()
mergedDf = pd.merge(departures_per_airport, us_airports_df, left_on='ORIGIN_AIRPORT', right_on='iata_code')
mergedDf = mergedDf.drop(columns=['latitude_deg', 'longitude_deg', 'region_name', 'municipality',"name","iso_region"])
mergedDf = mergedDf.groupby(['local_region']).sum().reset_index()

mergedDf["count"] = (mergedDf["count"] / 365).astype(int)

fig = px.choropleth(
    locations=mergedDf["local_region"], 
    locationmode="USA-states",
    color=mergedDf["count"],
    color_continuous_scale="brwnyl",    # blues, brwnyl
    range_color=(mergedDf["count"].min(), mergedDf["count"].max()),
    scope="usa",
    title="Departures from each US state per day"
)

# Don't show lake holes
fig.update_layout(geo=dict(showlakes=False), coloraxis_colorbar=dict(title="Daily departures"))

fig.update_layout(
    geo_scope='usa',
    width=1200,
    height=800
)

fig.show(renderer="png")
